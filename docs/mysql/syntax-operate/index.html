LEFT JOIN `sys_user_role` sur ON su.ID = sur.UserId
LEFT JOIN `sys_role` sr ON sur.UserRoleId = sr.id AND sr.isDelete = 0
LEFT JOIN `t_company` com ON com.id = su.CompanyId AND com.IsDelete = 0
LEFT JOIN `sys_user` a ON a.id = su.UpdateManId
LEFT JOIN `sys_user` b ON b.id = su.UpdateManId;

SELECT * FROM v_sysuser;
