MySQL存储过程示例

DROP PROCEDURE IF EXISTS `proc_appunionweight`;
DELIMITER $$
CREATE PROCEDURE `proc_appunionweight`(v_apitype VARCHAR(50))
BEGIN
    SELECT
        CASE
        WHEN apptypeid <> '' AND appqid <> '' AND slotid <> '' THEN CONCAT_WS('_',apptypeid,appqid,slotid)
        WHEN apptypeid <> '' AND appqid = '' AND slotid <> '' THEN CONCAT_WS('_',apptypeid,slotid)
        WHEN apptypeid = '' AND appqid = '' AND slotid <> '' THEN slotid
        END AS configtypeid,
    apptypeid,
    appver AS configversion,
    os,
    weightver AS version,
    uniontype AS configname,
    weight AS value,
    packagepath,
    issdk
    FROM adplatform_app_slotid  WHERE enable = 'Y' AND enableweight='Y' AND FIND_IN_SET(v_apitype,apitype) ORDER BY configtypeid DESC,weightver DESC;
    
END $$
DELIMITER ;

带入参和返回值

DROP PROCEDURE IF EXISTS `proc_editbiddinginfo`;
DELIMITER $$
CREATE PROCEDURE `proc_editbiddinginfo`(varplatform varchar(20),varqid VARCHAR(50),varlocationtype VARCHAR(50),varcpm VARCHAR(20),
              varsourcetype VARCHAR(50),varimpression VARCHAR(20),varclick VARCHAR(20),varincome VARCHAR(20),varctr VARCHAR(20),varcpc VARCHAR(20))
BEGIN
 DECLARE sp_error INT DEFAULT 0; 
 DECLARE result INT DEFAULT 0; 
 DECLARE CONTINUE HANDLER FOR SQLEXCEPTION SET sp_error=1; 
 IF varplatform = 'app' THEN 
    UPDATE adver_position_cpm_for_bid_by_app
    SET cpm = varcpm,sourcetype=varsourcetype,impression=varimpression,click=varclick,income=varincome,ctr=varctr,cpc=varcpc
    WHERE 
    qid = varqid AND locationtype = varlocationtype;
    SET result = 1;
 ELSEIF varplatform = 'h5' THEN 
    IF varqid = 'other' THEN
      UPDATE adver_position_cpm_for_bid_fake
      SET cpm = varcpm,sourcetype=varsourcetype,impression=varimpression,click=varclick,income=varincome,ctr=varctr,cpc=varcpc
      WHERE 
      qid = varqid AND locationtype = varlocationtype;
    ELSE
      UPDATE adver_position_cpm_for_bid_qid
      SET cpm = varcpm,sourcetype=varsourcetype,impression=varimpression,click=varclick,income=varincome,ctr=varctr,cpc=varcpc
      WHERE 
      qid = varqid AND locationtype = varlocationtype;
    END IF;
    SET result = 1;
 ELSEIF varplatform = 'pc' THEN 
    UPDATE adver_position_cpm_for_bid_fake
    SET cpm = varcpm,sourcetype=varsourcetype,impression=varimpression,click=varclick,income=varincome,ctr=varctr,cpc=varcpc
    WHERE 
    qid = varqid AND locationtype = varlocationtype;
    SET result = 1;
 END IF;
 IF sp_error=1 THEN
    SELECT sp_error;
    ROLLBACK;
 ELSE
    SELECT result;
    COMMIT;
 END IF;
END $$
DELIMITER ;
