})
public void updateEmployeeName(String employeeId, String lastName) {
    Optional<Employee> optional = this.findByEmployeeId(employeeId);
    if (optional.isPresent()) {
        Employee employee = optional.get();
        employee.setLastName(lastName);
        employeeRepository.save(employee);
    }
}
